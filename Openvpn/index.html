Configuration

    port 8080

    proto tcp

    dev tun

    ca /pki/ca.crt

    cert /pki/issued/Sserver.crt

    key /pki/private/Sserver.key

    dh /pki/dh.pem

    server 10.8.0.0 255.255.255.0

    ifconfig-pool-persist ipp.txt

    client-to-client

    comp-lzo

    persist-key

    persist-tun

    chroot /pki/conf/

    client-config-dir ccd

    verb 3

    client

    remote 192.168.2.45 8080

    proto tcp

    dev tun

    ca /pki/ca.crt

    pkcs12 "/pki/private/Dominique.p12"

    comp-lzo

    persist-key

    persist-tun
